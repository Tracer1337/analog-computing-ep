\newglossaryentry{dnn}{name={DNN},description={Deep Neuronal Network; Neuronales Netz mit vielen versteckten Ebenen}}

\newglossaryentry{rnn}{name={RNN},description={Recurrent Neural Network; Neuronales Netz mit rückwärtigen Verbindungen, Hauptbestandteil ist hier die Speicherzelle}}

\newglossaryentry{forward_pass}{name={Forward Pass},description={Erster Schritt im \gls{backpropagation}, in dem die Trainingsdaten das Modell durchlaufen}}
\newglossaryentry{reverse_pass}{name={Reverse Pass},description={Zweiter Schritt im \gls{backpropagation}, in dem das \gls{gradient_descent} auf die Gewichtungen zwischen Neuronen angewendet wird}}
